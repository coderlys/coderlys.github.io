    import axios from "axios";
import { BASE_URL, TIME_OUT } from "./config";  // 在配置文件中导入配置

import { storeToRefs } from "pinia";
import useStoreMain from "@/store/modules/main";
let storeMain = useStoreMain()
let { loadimg} = storeToRefs(storeMain) // 主要导入 加载loadimg的显示或隐藏

class Request {
    
  constructor(baseUrl, timeOut) {
    this.instance = axios.create({
      baseURL: baseUrl,
      timeout: timeOut,
    });

    this.instance.interceptors.request.use((config)=>{ // 请求拦截器
      loadimg.value = true 
      return config 
    })

    this.instance.interceptors.response.use((res)=>{ // 相应拦截器
      loadimg.value = false
      return res
    },(error)=>{
      loadimg.value = false;
      console.log(error);
    })
  }

  request(config) {
    return new Promise((resolve, reject) => {
      this.instance.request({ ...config })
        .then((res) => {
          resolve(res.data);
        })
        .catch((error) => {
          reject(error);
        });
    });
  }

  get(config) {
    console.log(config);
    return this.request({ ...config, method: "get" });
  }
  post(config) {
    return this.request({ ...config, method: "post" });
  }
}

export default new Request(BASE_URL, TIME_OUT);

    # 五：列表加载更多数据

    封装的 hook / useScroll.js文件

    import { onMounted, onUnmounted , ref} from "vue";
import { throttle } from "underscore";
export const useScroll = (elRef)=>{
    let el = document
    let scrollState = ref(false)  // 是否滚动已经到底部
    let clientH = ref(0) //  页面的高度
    let scrollT = ref(0) //  已滚动的高度
    let scrollH= ref(0)  //  整体页面的高度

    const scrollFunc = throttle(()=>{
        if(elRef){
            clientH.value = el.clientHeight; // 当前页面高度
            scrollT.value = Math.ceil(el.scrollTop) // 已滚动的高度
            scrollH.value = el.scrollHeight // 整体页面的高度
        }
        else{
            clientH.value = document.documentElement.clientHeight; // 当前页面高度
            scrollT.value = Math.ceil(document.documentElement.scrollTop) // 已滚动的高度
            scrollH.value = document.documentElement.scrollHeight // 整体页面的高度
        }
        if(scrollH.value<=scrollT.value+clientH.value){
            scrollState.value = true
        }
    },100) 
    onMounted(()=>{  // 生命周期 组件挂载的时候
        if(elRef) el = elRef.value
        console.log(el);
        el.addEventListener('scroll',scrollFunc)
    })
    onUnmounted(()=>{ // 生命周期 组件销毁的时候
        el.removeEventListener('scroll',scrollFunc)
    })
    return {scrollState,clientH,scrollT}
}

    const detailsItemEl = []
const getdetailsItemTop = ((value)=>{ // 组件的 绑定的函数会自动触发
    if(!value) return // 返回退出的时候，会触发value 之前会 null 所以需要判断下 ，不然会报错
    detailsItemEl[value.$el.getAttribute('name')] = value.$el  // 获取对应的数据内容 例如 描述:el
})
const names = computed(()=>{
    return Object.keys(detailsItemEl)  // 获取所有 tab切换栏的名字
})


let isclick = false  // 控制是否点击tab的切换
let scrollHeight = 0 // 获取点击滚动的高度

const changeTop = (index)=>{ // 点击 tab切换栏的名字 ，滚动到对应的位置

let key = Object.keys(detailsItemEl)[index];
const el = detailsItemEl[key]; // 获取被点击的元素
let distance = el.offsetTop  // 获取该元素的top的距离

scrollHeight = distance
if(index!=0){ // 第一个元素不需要 -40的tab的高度
	distance-=40
}
details.value.scrollTo({  // 设置top的高度
    top:distance, 
    behavior:'smooth' // 动画的滚动
})
isclick = true
}


// 监听滚动
watch(scrollT,(value)=>{
    if(value==scrollHeight){ // 如果已经滚动到指定的位置，就进行解锁
        isclick = false 
    }
    if(!isclick){  // 如果当前是被点击的话，就不进行以下的滚动
        let els = Object.values(detailsItemEl) // 获取所有的demo元素
        let tops= els.map((item)=>item.offsetTop)  // 获取所有的元素距离顶部的位置  [272, 515, 908, 1236, 1559, 1834]
        let index = tops.length-1 // 超过的元素滚动的距离，则选择最后一个
        for (let i = 0; i < tops.length; i++) {
            if(tops[i]>value+40){
                index = i-1; // 选择前一个索引
                break
            }
        }
        tabIndex.value = index 
    }
})
